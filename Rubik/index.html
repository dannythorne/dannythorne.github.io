switch(key)
  {
    case 'r': // right face clockwise
    case 'R': // right face counter-clockwise
    case 'l': // left face clockwise
    case 'L': // left face counter-clockwise
    case 'u': // up face clockwise
    case 'U': // up face counter-clockwise
    case 'd': // down face clockwise
    case 'D': // down face counter-clockwise
    case 'f': // front face clockwise
    case 'F': // front right face counter-clockwise
    case 'b': // back face clockwise
    case 'B': // back face counter-clockwise
    case 'm': // middle (must be followed by one of the above)
    case 'M': // middle on until pressed again
    case 'x': // clockwise around x-axis (whole cube)
    case 'y': // clockwise around y-axis (whole cube)
    case 'z': // clockwise around z-axis (whole cube)
    case 'X': // counter-clockwise around x-axis (whole cube)
    case 'Y': // counter-clockwise around y-axis (whole cube)
    case 'Z': // counter-clockwise around z-axis (whole cube)
    case 's': // one random turn (press repeatedly to scramble cube)
    case ' ': if( shift_is_on) { /*zoom out*/ } else { /*zoom in*/}
    case DELETE: // undo
    case BACKSPACE: // undo
    case TAB: // redo
  }
